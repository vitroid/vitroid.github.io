    import itertools as it
    import random
    
    a = "アマ"
    n = "ソリン"
    pa = "パバペ"
    
    words = [i for i in it.product(a,n,pa,n,a,n)]
    random.shuffle(words)
    L = len(words) // 9
    rows = ["".join(row) for row in zip(*words)]
    
    for i in range(9):
        for j in range(6):
            r = rows[j]
            print(r[i*L:(i+1)*L])
        print()
